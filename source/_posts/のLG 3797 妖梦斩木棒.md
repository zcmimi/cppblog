#define N 200011
#define ls rt<<1
#define rs rt<<1|1
int n,q,s[N<<2],sl[N<<2],sr[N<<2];
void pu(int rt){
	s[rt]=s[ls]+s[rs];
	if(sl[ls]&&sr[rs])s[rt]++;
	if(sl[rs])sl[rt]=sl[rs];
	else if(!s[rs]&&!sr[rs]&&!sl[rs]&&sl[ls])sl[rt]=sl[ls];
	else sl[rt]=0;
	if(sr[ls])sr[rt]=s[ls];
	else if(!s[ls]&&!sl[ls]&&!sr[ls]&&sr[rs])sr[rt]=sr[rs];
	else sr[rt]=0;
}
void build(int l,int r,int rt){
	if(l==r){
		if(l==1)sl[rt]=1;
		if(l==n)sr[rt]=n;
		return;

	int m=(l+r)>>1;
	build(l,m,ls);
	build(m+1,r,rs);
	pu(rt);

void upd(int x,int c,int l,int r,int rt){

}
