#define N 2000011

const int mod = 19930726;
int n,ans=1,cnt[N];
ull h[N],H[N],pw[N];

il ull g(int l,int r){
    return h[r]-h[l-1]*pw[r-l+1];
}
il ull G(int l,int r){
    return H[l]-H[r+1]*pw[r-l+1];
}
il void ask(int x){
    int l=1,r=MIN(x,n-x);
    while(l<=r) {
        int m=(l+r)>>1;
        if(g(x-m,x+m)==G(x-m,x+m))
            l=m+1;
        else r=m-1;
    }
    cnt[r<<1|1]++;
}
il int px(int x,ll p){
    int ans=1;
    while(p){
        if(p&1)ans=(1ll*ans*x)%mod;

    }
    return ans;
