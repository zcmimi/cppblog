把站位当作抵消的部分

建立两棵线段树,统计$x$轴和$y$轴被覆盖的情况

每次的答案就是

放过的行数×行长度+放过的列数×列长度-抵消块数

抵消块数就是$x$轴被覆盖行数$\times y$轴被覆盖行数 $\times 2$

#define N 100011
#define ls rt<<1
#define rs rt<<1|1
int P[N<<2],Q[N<<2],n,m;
void upd(int *s,int x,int l,int r,int rt){
	if(l==r){
		s[rt]^=1;
		return;
	}
	int m=(l+r)>>1;
	if(x<=m)upd(s,x,l,m,ls);
	else upd(s,x,m+1,r,rs);
	s[rt]=s[ls]+s[rs];
}
int ask(int *s,int L,int R,int l,int r,int rt){
	if(L<=l&&r<=R)return s[rt];
	int m=(l+r)>>1,ans=0;
	if(L<=m)ans=ask(s,L,R,l,m,ls);
	if(R>m)ans+=ask(s,L,R,m+1,r,rs);
	return ans;
}
